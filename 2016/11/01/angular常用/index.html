推荐写法

myApp.directive('helloWorld',function(){
    return{
        templateUrl:"",//模板的url
        replace:true,//是否替换helloWrold标签
        scope:true, //是否引入父controller默认true scope可以写成{}断链写成false不能断链
        {
            message:'@message' //可以选择取父对象的某些属性 需在元素上加上一个属性message={{message}}  可以用来实现单向数据绑定
            // =等号可以实现双向数据绑定
        }
        controller:function(){

            }//定义自己的controller
        },
        compile:function(elem,attrs){

        }//compile返回一个link 如果return 就不需要写link 把link内容写在compile中
        ,
        link:function(scope,elem,attrs){
            //elem是一个jq对象
            //scope 可以拿到当前作用域
        }//可以拿到元素对象 属性集合 可以用来深度定制属性 属性里面-写法 取值用驼峰
        //angular 内置jqlite
    })
    angular解析模板 先把模板装载到一个节点上(compile) 渲染模板(link)

使用

配置

myApp.config(['$stateProvider','$urlRouterProvider','$urlMatcherFactoryProvider',function($stateProvider,$urlRouterProvider,$urlMatcherFactoryProvider){
    //$urlRouterProvider  otherwise挂在这个上面设置默认页面 需要注入$urlRouterProvider服务
    //$urlMatcherFactoryProvider.caseInsensitiveMatch() //设置路由对大小写是否敏感
    $stateProvider
        .state('home',{//路由的名字
            url:'/home', //真正的路由 如果传参'/home/:参数名'
            templateUrl:'',//路由的路径
            controller:"",//定义控制器
            controllerAs:'',
            resolve:{ //延迟执行
                employeelist:function(){

                }
            }
            })
    $urlRouterProvider.otherwise('/home');//设置默认页面 可以单独设置一个config抽离
    $urlMatchFactoryProvider.caseInsensitiveMatch(true);//设置对路由大小写敏感
    }])
