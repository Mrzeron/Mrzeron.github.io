super可以在子对象里面直接访问父对象的方法和属性

let breakfast={
    getDrink(){
        return 'tea';
    }
}
let dinner={
    getDrink(){
        return 'bear';
    }
}
let sunday ={
    __proto__:breakfast;
    getDrink(){
        return super.getDrink();
    }
}
console.log(sunday.getDrink())

class 类

class Chef{
    constructor(food){
        this.food=food;
        this.dish=[];
    }
    cook(){
        console.log(this.food)
    }
    get menu(){  //get+空格+方法名
        return this.dish;
    }
    set menu(dish){
        this.dish.push(dish)
    }
}
let gp = new Chef('tomato');
//gp.cook();
gp.menu ="pizza";
gp.menu ="apple";
console.log(gp.menu) //后面不带等号调用get方法

静态方法

class Chef{
    constructor(food){
        this.food=food;
        this.dish=[];
    }
    static cook(food){      //静态方法
        console.log(food)
    }
    get menu(){  //get+空格+方法名
        return this.dish;
    }
    set menu(dish){
        this.dish.push(dish)
    }
}    
Chef.cook('tomato') //静态方法可以直接用类名调用不报错

继承

class Person{
    constructor(name,birthday){
        this.name=name;
        this.birthday=birthday;
    }
    intro(){
        return `${this.name},$(this.birthday)`;
    }
}    
class Chef extends Person{
    constructor(name,birthday){
        super(name,birthday)
    }

}
let gp =new Chef("pz",'1992-02-21');
console.log(gp.intro())
