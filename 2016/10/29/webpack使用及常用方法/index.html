webpack使用及常用方法

发表于 2016-10-29   |     |   阅读次数

webpack实现前端工程化七项功能

  1. cnpm i webpack -g 全局安装webpack
  2. 成功 webpack –help
  3. cnpm init

  4. cnpm i webpack -D 本地安装
    新建entry.js index.html

    webpack entry.js bundle.js 编译entry.js为bundle.js
    支持commonJS规范 module.exports={} require

  5. cnpm i css-loader style-loader -D
    require(‘style!css!./style.css’);

    配置文件

    新建webpack.config.js
    module.exports={

    entry:'./entry.js',//入口
output:{
path:__dirname, //当前文件夹
filename:'bundle.js' //文件名字
}，
devtool:'source-map',//配置source-map        
module:{
loaders:[
    {
        test:/\.css$/,
        loader:'style!css' //style和css位置不能交换
    }    
]    //配置模块
}

    }

    打开页面输入webpack
    打开source-map
    webpack –devtool source-map
    安装server
    cnpm i webpack-dev-server -g //全局安装
    cnpm i webpack-dev-server -D //本地安装
    webpack-dev-server –inline –hot
    需要输入localhost:8080才能访问
    cnpm i babel-loader babel-core babbel-preset-es2015 babbel-preset-react -D

配置入口
entry:__dirname+'/src/scripts/app.js',
配置map
devtool:'source-map',
配置出口
output:{
    path:__dirname+'/prd/',
    filename:'bundle.js'
}

配置server

devServer: {
contentBase: __dirname + '/prd',
port: 80,
inline: true

},

sass编译

cnpm i sass-loader node-sass -D
配置sass
module.exports={
    entry:'./entry.js',//入口
    output:{
    path:__dirname, //当前文件夹
    filename:'bundle.js' //文件名字
    }，
    devtool:'source-map',//配置source-map        
    module:{
    loaders:[
        {
            test:/\.css$/,//改了这一行
            loader:'style!css'  //改了这一行
        },
        {
            test:/\.scss$/,//改了这一行
            loader:'style!css!sass'  //改了这一行
        }    


    ]    //配置模块
    }
}
使用->import ''

编译es6

cnpm i babel-loader babel-core babbel-preset-es2015 babbel-preset-react -D
配置preset
创建.babelrc
{
    "presets":["es2015"]
}
配置loader
//数组中添加一项
loaders:[
        {
            test:/\.js$/,
            loader:'babel'  //改了这一行
        }    
    ]

抽离

cnpm i extract-text-webpack-plugin -D
配置config
var ET =require('extract-text-webpack-plugin');
{
            test:/\.scss$/,
            loader:ET.extract('style','css!sass')
        }    

//plugin定义和module平级
plugins:[
new    ET('bundle.css') //目标文件夹 参照上面出口
]        
//打包是webpack做的

修改配置后server必须重启

压缩

配置config
var webpack =require()
plugins:[
new webpack.optimize.UglifyJsplugin() //发布时在用
]

版本号

output:{
    path:__dirname+'/prd',
    filename:'[name]-[hash].js'
}    
    plugins:[
new    ET(__dirname + '[name]-[hash].css') //目标文件夹 参照上面出口
]

mock数据

cnpm i json-server -g
配置 mock.js 
var fri_msg=require ("./fri_msg.js");
    module.exports=function(){
        return {
            "fri_msg":fri_msg
        }
    }
配置fri_msg.js
module.exports= [{
                img:"http://ofltwdk94.bkt.clouddn.com/Teemo.png",
                name:"炒鸡困要睡觉",
                state:"游戏离线"
            },{
                    img:"http://ofltwdk94.bkt.clouddn.com/Teemo.png",
                name:"红烧鸡翅膀",
                state:"游戏离线"
            }]


执行 json-server mock/mock.js -w -p 端口
修改webpack.config.js
在devServer里面配置代理-> proxy:{
    '/rest/*'：{
        target:    'http://localhost:端口号',
        secure:false,
        pathRewrite:{
            '^/rest':''   //以rest开头的替换为空 /rest/list->localhost:8888/list
        }
        }    //前面以/rest/开头的所有都使用代理转发
}
